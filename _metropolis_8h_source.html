289  K_, Lambda_);

295  K_, Lambda_);

301  K_, Lambda_);

476  void attempt_move(const move_type move)
477  {
478  // Calculate probability
479  auto a1 = CalculateA1(move);
480  // Make move if random number < probability
481  auto a2 = CalculateA2(move);
482 
483  const auto trial_value = generate_probability();
484  // Convert to Gmpzf because trial_value will be set to 0 when
485  // comparing with a1 and a2!
486  // const auto trial = Gmpzf(static_cast<double>(trial_value));
487  const auto trial = static_cast<double>(trial_value);
488 
489 #ifndef NDEBUG
490  std::cout << __PRETTY_FUNCTION__ << " called." << std::endl;
491  std::cout << "trial_value = " << trial_value << std::endl;
492  std::cout << "trial = " << trial << std::endl;
493 #endif
494 
495  if (trial <= a1 * a2)
496  {
497  // Move accepted
498  make_move(move);
499  }
500  else
501  {
502  // Move rejected
503  // Increment attempted_moves_
504  ++attempted_moves_[to_integral(move)];
505  }
506 
507 #ifndef NDEBUG
508  std::cout << __PRETTY_FUNCTION__ << " called." << std::endl;
509  std::cout << "Attempting move." << std::endl;
510  std::cout << "Move type = " << to_integral(move) << std::endl;
511  std::cout << "Trial = " << trial << std::endl;
512  std::cout << "A1 = " << a1 << std::endl;
513  std::cout << "A2 = " << a2 << std::endl;
514  std::cout << "A1*A2 = " << a1 * a2 << std::endl;
515  std::cout << ((trial <= a1 * a2) ? "Move accepted." : "Move rejected.")
