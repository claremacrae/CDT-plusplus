43 /// @brief Metropolis-Hastings algorithm function object

45 /// The Metropolis-Hastings algorithm is a Markov Chain Monte Carlo method.
46 /// The probability of making an ergodic (Pachner) move is:
47 ///
48 /// \f[P_{ergodic move}=a_{1}a_{2}\f]
49 /// \f[a_1=\frac{move[i]}{\sum\limits_{i}move[i]}\f]
50 /// \f[a_2=e^{\Delta S}\f]

114  /// @brief Calculate A1
115  ///
116  /// Calculate the probability of making a move divided by the
117  /// probability of its reverse, that is:
118  /// \f[a_1=\frac{move[i]}{\sum\limits_{i}move[i]}\f]
119  ///
120  /// @param move The type of move
121  /// @return \f$a_1=\frac{move[i]}{\sum\limits_{i}move[i]}\f$
122  auto CalculateA1(const move_type move) const noexcept
123  {
124  auto total_moves = this->TotalMoves();
125  auto this_move = attempted_moves_[to_integral(move)];
126  // Set precision for initialization and assignment functions
127  mpfr_set_default_prec(PRECISION);
128 
129  // Initialize for MPFR
130  mpfr_t r1, r2, a1;
131  mpfr_inits2(PRECISION, r1, r2, a1, nullptr);
132 
133  mpfr_init_set_ui(r1, this_move, MPFR_RNDD); // r1 = this_move
134  mpfr_init_set_ui(r2, total_moves, MPFR_RNDD); // r2 = total_moves
135 
136  // The result
137  mpfr_div(a1, r1, r2, MPFR_RNDD); // a1 = r1/r2
138 
139  // std::cout << "A1 is " << mpfr_out_str(stdout, 10, 0, a1, MPFR_RNDD)
140

143  // MP_Float result = MP_Float(mpfr_get_ld(a1, MPFR_RNDD));
144  auto result = mpfr_get_d(a1, MPFR_RNDD);
145 
146  // Free memory
147  mpfr_clears(r1, r2, a1, nullptr);

157  /// @brief Calculate A2

159  /// Calculate \f$a_2=e^{\Delta S}\f$

161  /// @param move The type of move
162  /// @return \f$a_2=e^{-\Delta S}\f$
163  auto CalculateA2(const move_type move) const noexcept
164  {
165  auto currentS3Action =
166  S3_bulk_action(N1_TL_, N3_31_13_, N3_22_, Alpha_, K_, Lambda_);
167  auto newS3Action = static_cast<Gmpzf>(0);
168  // auto newS3Action = static_cast<MP_Float>(0);
169  switch (move)
170  {
171  case move_type::TWO_THREE:
172  // A (2,3) move adds a timelike edge
173  // and a (2,2) simplex
174  newS3Action = S3_bulk_action(N1_TL_ + 1, N3_31_13_, N3_22_ + 1, Alpha_,
175  K_, Lambda_);
176  break;
177  case move_type::THREE_TWO:
178  // A (3,2) move removes a timelike edge
179  // and a (2,2) simplex
180  newS3Action = S3_bulk_action(N1_TL_ - 1, N3_31_13_, N3_22_ - 1, Alpha_,
181  K_, Lambda_);
182  break;
183  case move_type::TWO_SIX:
184  // A (2,6) move adds 2 timelike edges and
185  // 2 (1,3) and 2 (3,1) simplices
186  newS3Action = S3_bulk_action(N1_TL_ + 2, N3_31_13_ + 4, N3_22_, Alpha_,
187  K_, Lambda_);
188  break;
189  case move_type::SIX_TWO:
190  // A (6,2) move removes 2 timelike edges and
191  // 2 (1,3) and 2 (3,1) simplices
192  newS3Action = S3_bulk_action(N1_TL_ - 2, N3_31_13_, N3_22_ - 4, Alpha_,
193  K_, Lambda_);
194  break;
195  case move_type::FOUR_FOUR:
196 // A (4,4) move changes nothing with respect to the action,
197 // and e^0==1
198 #ifndef NDEBUG

200 #endif
201  return static_cast<double>(1);
202  }

239  /// @brief Attempt a move of the selected type
240  ///
241  /// This function implements the core of the Metropolis-Hastings algorithm
242  /// by generating a random number and comparing with the results of
243  /// CalculateA1 and CalculateA2. If the move is accepted, this function
244  /// calls make_move(). If not, it updates **attempted_moves_**.
245  ///
246  /// @param move The type of move
247  void attempt_move(const move_type move)
248  {
249  // Calculate probability
250  auto a1 = CalculateA1(move);
251  // Make move if random number < probability
252  auto a2 = CalculateA2(move);
253 
254  const auto trial_value = generate_probability();
255  // Convert to Gmpzf because trial_value will be set to 0 when
256  // comparing with a1 and a2!
257  // const auto trial = Gmpzf(static_cast<double>(trial_value));
258  const auto trial = static_cast<double>(trial_value);
259 
260 #ifndef NDEBUG
261  std::cout << __PRETTY_FUNCTION__ << " called.\n";
262  std::cout << "trial_value = " << trial_value << "\n";
263  std::cout << "trial = " << trial << "\n";
264 #endif
265 
266  if (trial <= a1 * a2) {
267  // Move accepted
268  make_move(move);
269  }
270  else
271  {
272  // Move rejected
273  // Increment attempted_moves_
274  ++attempted_moves_[to_integral(move)];
275  }
276 
277 #ifndef NDEBUG
278  std::cout << __PRETTY_FUNCTION__ << " called.\n";
279  std::cout << "Attempting move.\n";
280  std::cout << "Move type = " << to_integral(move) << "\n";
281  std::cout << "Trial = " << trial << "\n";
282  std::cout << "A1 = " << a1 << "\n";
283  std::cout << "A2 = " << a2 << "\n";
284  std::cout << "A1*A2 = " << a1 * a2 << "\n";
285  std::cout << ((trial <= a1 * a2) ? "Move accepted." : "Move rejected.")
286  << "\n";
287  print_run();
288 #endif
289  } // attempt_move()
