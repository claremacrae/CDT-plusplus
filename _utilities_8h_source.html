313 inline auto expected_points_per_simplex(const int dimension,

323 
324  const auto simplices_per_timeslice = simplices / timeslices;
325  switch (dimension)
326  {
327  case 3:
328  {
329  // Avoid segfaults for small values

332  return 4 * simplices_per_timeslice;

336  return simplices_per_timeslice;
337  }

341  }

345  }
346  }

350  }
