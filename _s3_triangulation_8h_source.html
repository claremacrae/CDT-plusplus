128 #ifndef NDEBUG
129  std::cout << "Classifying edges...." << std::endl;

131  Delaunay::Finite_edges_iterator eit;
132  std::vector<Edge_handle> timelike_edges;
133  std::vector<Edge_handle> spacelike_edges;
134 
135  // Iterate over all edges in the Delaunay triangulation
136  for (eit = universe_ptr->finite_edges_begin();
137  eit != universe_ptr->finite_edges_end(); ++eit)
138  {
139  Cell_handle ch = eit->first;
140  // Get timevalues of vertices at the edge ends
141  auto time1 = ch->vertex(eit->second)->info();
142  auto time2 = ch->vertex(eit->third)->info();
143 
144  // Make Edge_handle
145  Edge_handle thisEdge{
146  ch, static_cast<std::intmax_t>(ch->index(ch->vertex(eit->second))),
147  static_cast<std::intmax_t>(ch->index(ch->vertex(eit->third)))};
148 
149  if (time1 != time2)
150  { // We have a timelike edge
151  timelike_edges.emplace_back(thisEdge);
152 
153 #ifdef DETAILED_DEBUGGING
154  std::cout << "First vertex of edge is " << std::get<1>(thisEdge)
155  << " and second vertex of edge is " << std::get<2>(thisEdge)
156  << std::endl;
157 #endif
158  }
159  else
160  { // We have a spacelike edge
161  spacelike_edges.emplace_back(thisEdge);
162  } // endif
163  } // Finish iterating over edges
164 
165 // Display results if debugging

167  std::cout << "There are " << timelike_edges.size() << " timelike edges and "

200  // Iterate over all cells in the Delaunay triangulation

202  cit != universe_ptr->finite_cells_end(); ++cit)
203  {
204  std::intmax_t max_values{0};
205  std::intmax_t min_values{0};
206  // Push every time value of every vertex into a list
207  std::intmax_t timevalues[4] = {

209  cit->vertex(3)->info(),
210  };
211  std::intmax_t max_time =
212  *std::max_element(std::begin(timevalues), std::end(timevalues));
213  for (auto elt : timevalues)
214  {

216  {
217  ++max_values;
218  }
219  else
220  {
221  ++min_values;
222  }
223  }
224 
225  // Classify simplex using max_values and write to cit->info()
226  if (min_values == 1 && max_values == 3)
227  {
228  cit->info() = 13;
229  one_three.emplace_back(cit);
230  }
231  else if (min_values == 2 && max_values == 2)

235  }

293 auto fix_timeslices(T&& universe_ptr)
294 { // NOLINT
295  Delaunay::Finite_cells_iterator cit;
296  std::intmax_t min_time{0};
297  std::intmax_t max_time{0};
298  std::intmax_t valid{0};
299  std::intmax_t invalid{0};
300  std::intmax_t max_vertex{0};
301  std::set<Vertex_handle> deleted_vertices;
302 
303  // Iterate over all cells in the Delaunay triangulation
304  for (cit = universe_ptr->finite_cells_begin();
305  cit != universe_ptr->finite_cells_end(); ++cit)
306  {
307  if (cit->is_valid())
308  { // Valid cell
309  min_time = cit->vertex(0)->info();
310  max_time = min_time;
311 #ifdef DETAILED_DEBUGGING
312  bool this_cell_foliation_valid = true;
313 #endif
314  // Iterate over all vertices in the cell
315  for (auto i = 0; i < 4; ++i)
316  {
317  auto current_time = cit->vertex(i)->info();
318 
319  // Classify extreme values
320  if (current_time < min_time) min_time = current_time;
321  if (current_time > max_time)
322  {
323  max_time = current_time;
324  max_vertex = static_cast<intmax_t>(i);
325  }
326  } // Finish iterating over vertices
327  // There should be a difference of 1 between min_time and max_time
328  if (max_time - min_time != 1)
329  {
330  invalid++;
331 #ifdef DETAILED_DEBUGGING
332  this_cell_foliation_valid = false;
333 #endif
334  // Single-threaded delete max vertex
335  // universe_ptr->remove(cit->vertex(max_vertex));
336 
337  // Parallel delete std::set of max_vertex for all invalid cells
338  deleted_vertices.emplace(cit->vertex(static_cast<int>(max_vertex)));
339  }
340  else
341  {
342  ++valid;
343  }
344 
345 #ifdef DETAILED_DEBUGGING
346  std::cout << "Foliation for cell is "
347  << ((this_cell_foliation_valid) ? "valid." : "invalid.")
348  << std::endl;
349  for (auto i = 0; i < 4; ++i)
350  {
351  std::cout << "Vertex " << i << " is " << cit->vertex(i)->point()
352  << " with timeslice " << cit->vertex(i)->info() << std::endl;
353  }
354 #endif
355  }
356  else
357  {
358  throw std::runtime_error("Cell handle is invalid!");
359  // Or just remove the cell
360  // universe_ptr->tds().delete_cell(cit);
361  // This results in a possibly broken Delaunay triangulation
362  // Or possibly just delete a vertex in the cell,
363  // perhaps forcing a re-triangulation?
364  }
365  } // Finish iterating over cells
366 
367  // Delete invalid vertices
368  universe_ptr->remove(deleted_vertices.begin(), deleted_vertices.end());
369  // Check that the triangulation is still valid
370  CGAL_triangulation_expensive_postcondition(universe_ptr->is_valid());
371 
372 #ifndef NDEBUG
373  std::cout << "There are " << invalid << " invalid simplices and " << valid
374  << " valid simplices." << std::endl;
375 #endif
376  return invalid == 0;
377 } // fix_timeslices
